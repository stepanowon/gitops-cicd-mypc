# Namespace 생성, Helm 리포지토리 등록
kubectl create namespace haproxy-controller
helm repo add haproxytech https://haproxytech.github.io/helm-charts
helm repo update

# haproxy-ingress-controller 설치, prometheus 관련 설정 포함 
helm upgrade -i -n haproxy-controller haproxy-ingress haproxytech/kubernetes-ingress \
  --set controller.service.type=LoadBalancer \
  --set controller.service.loadBalancerIP=192.168.56.60 \
  --set controller.ingressClass=haproxy \
  --set controller.ingressClassResource.name=haproxy \
  --set controller.ingressClassResource.enabled=true \
  --set controller.metrics.enabled=true \
  --set controller.metrics.service.type=ClusterIP \
  --set controller.metrics.service.port=1024 \
  --set-string controller.podAnnotations."prometheus\.io/scrape"="true" \
  --set-string controller.podAnnotations."prometheus\.io/port"="1024" \
  --set-string controller.podAnnotations."prometheus\.io/path"="/metrics"

# Prometheus server 설치
kubectl apply -f \
   https://raw.githubusercontent.com/stepanowon/gitops-cicd-mypc/refs/heads/main/prometheus-server.yaml
